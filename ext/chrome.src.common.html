List listify(JsObject obj, [Function transformer = null]) {
 if (obj == null) {
   return null;
 } else {
   List l = new List(obj['length']);

   for (int i = 0; i < l.length; i++) {
     if (transformer != null) {
       l[i] = transformer(obj[i]);
     } else {
       l[i] = obj[i];
     }
   }

   return l;
 }
}
